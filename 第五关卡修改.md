```javascript
  function bindCase5(caseDef, cs) {
    if (cs.phase !== "deduction") {
      document.getElementById("btnToDeduce5")?.addEventListener("click", () => {
        beginPhase(cs, "deduction");
        renderCase(caseDef.id);
      });
      return;
    }

    const world = document.getElementById("jnjWorld");
    const fill = document.getElementById("jnjFill");
    const safeZone = document.getElementById("jnjSafeZone");
    const btnCharge = document.getElementById("btnJumpCharge");
    const stepNum = document.getElementById("jnjStepNum");
    const hud = document.getElementById("jnjHud");
    const terminal = document.getElementById("jnjTerminal");
    const stepsData = caseDef.data?.steps || [];

    // 游戏状态
    let currentStep = 0;
    let selectedTarget = null; // 'left' or 'right'
    let requiredPower = 0;
    let chargeValue = 0;
    let charging = false;
    let chargeAnimFrame = null;
    let isJumping = false;
    
    // 固定的 3D 空间坐标（基于中心点 0,0）
    const POS = {
      current: { x: 0, y: 0 },
      left: { x: -160, y: 0 },   // 左前方
      right: { x: 0, y: -160 }   // 右前方
    };

    function initStage() {
      if (!world) return;
      world.innerHTML = "";
      
      if (currentStep >= stepsData.length) {
        // 游戏结束，显示终端
        hud.classList.add("hidden");
        terminal.classList.remove("hidden");
        
        // 生成终点平台
        const endPlat = document.createElement("div");
        endPlat.className = "jnj-plat is-current";
        endPlat.style.transform = `translate3d(0px, 0px, 0)`;
        endPlat.innerHTML = `<div class="jnj-text">零点休息室</div>`;
        
        const pawn = document.createElement("div");
        pawn.className = "jnj-pawn";
        pawn.style.transform = `translate3d(0px, 0px, 0)`;
        
        world.appendChild(endPlat);
        world.appendChild(pawn);
        return;
      }

      stepNum.textContent = currentStep + 1;
      const stepInfo = stepsData[currentStep];

      // 生成当前立足点
      const curPlat = document.createElement("div");
      curPlat.className = "jnj-plat is-current";
      curPlat.style.transform = `translate3d(${POS.current.x}px, ${POS.current.y}px, 0)`;
      curPlat.innerHTML = `<div class="jnj-text">起跳点</div>`;
      world.appendChild(curPlat);

      // 生成左侧目标
      const leftPlat = document.createElement("div");
      leftPlat.className = "jnj-plat";
      leftPlat.style.transform = `translate3d(${POS.left.x}px, ${POS.left.y}px, 0)`;
      leftPlat.innerHTML = `<div class="jnj-text">${escapeHtml(stepInfo.left.id)}</div>`;
      leftPlat.onclick = () => selectTarget('left', leftPlat, POS.left);
      world.appendChild(leftPlat);

      // 生成右侧目标
      const rightPlat = document.createElement("div");
      rightPlat.className = "jnj-plat";
      rightPlat.style.transform = `translate3d(${POS.right.x}px, ${POS.right.y}px, 0)`;
      rightPlat.innerHTML = `<div class="jnj-text">${escapeHtml(stepInfo.right.id)}</div>`;
      rightPlat.onclick = () => selectTarget('right', rightPlat, POS.right);
      world.appendChild(rightPlat);

      // 生成玩家棋子
      const pawn = document.createElement("div");
      pawn.className = "jnj-pawn";
      pawn.id = "jnjPawn";
      pawn.style.transform = `translate3d(${POS.current.x}px, ${POS.current.y}px, 0)`;
      world.appendChild(pawn);

      // 重置状态
      selectedTarget = null;
      btnCharge.textContent = "👆 请先点击选择前方的平台";
      btnCharge.disabled = true;
      safeZone.style.width = "0%";
      fill.style.width = "0%";
    }

    function selectTarget(side, el, pos) {
      if (isJumping || charging) return;
      world.querySelectorAll('.jnj-plat').forEach(p => p.classList.remove('is-target'));
      el.classList.add('is-target');
      selectedTarget = { side, x: pos.x, y: pos.y };
      
      // 随机生成所需的力度 (40% - 85%)，增加游戏性
      requiredPower = 40 + Math.random() * 45; 
      
      // 设置绿色安全区 (容错率 ±8%)
      const safeWidth = 16;
      const safeLeft = requiredPower - (safeWidth / 2);
      safeZone.style.left = `${safeLeft}%`;
      safeZone.style.width = `${safeWidth}%`;
      
      btnCharge.disabled = false;
      btnCharge.textContent = "长按蓄力，绿色区松手！";
    }

    // --- 核心：蓄力与跳跃逻辑 ---
    const startCharge = (e) => {
      if (e) { e.preventDefault(); e.stopPropagation(); }
      if (isJumping || !selectedTarget || btnCharge.disabled) return;
      charging = true;
      chargeValue = 0;
      document.getElementById("jnjPawn")?.classList.add("charging");
      
      let lastTime = performance.now();
      const loop = (time) => {
        if (!charging) return;
        const delta = time - lastTime;
        lastTime = time;
        // 蓄力速度：每秒增长约 75%
        chargeValue += (delta * 0.075);
        if (chargeValue > 100) chargeValue = 100;
        fill.style.width = `${chargeValue}%`;
        chargeAnimFrame = requestAnimationFrame(loop);
      };
      chargeAnimFrame = requestAnimationFrame(loop);
    };

    const endCharge = (e) => {
      if (e) { e.preventDefault(); e.stopPropagation(); }
      if (!charging) return;
      charging = false;
      cancelAnimationFrame(chargeAnimFrame);
      const pawn = document.getElementById("jnjPawn");
      pawn?.classList.remove("charging");
      
      executeJump(chargeValue);
    };

    function executeJump(power) {
      isJumping = true;
      const pawn = document.getElementById("jnjPawn");
      
      // 判断是否落在安全区
      const tolerance = 8; 
      const isSuccess = Math.abs(power - requiredPower) <= tolerance;
      
      // 计算落点
      const endX = isSuccess ? selectedTarget.x : (selectedTarget.x * (power / requiredPower));
      const endY = isSuccess ? selectedTarget.y : (selectedTarget.y * (power / requiredPower));
      
      // 抛物线动画 (使用 Web Animations API)
      const jumpAnim = pawn.animate([
        { transform: `translate3d(0px, 0px, 0)` },
        { transform: `translate3d(${endX/2}px, ${endY/2}px, 120px)` }, // 最高点
        { transform: `translate3d(${endX}px, ${endY}px, ${isSuccess ? 0 : -50}px)` } // 落点
      ], {
        duration: 500,
        easing: 'cubic-bezier(0.25, 0.46, 0.45, 0.94)'
      });

      jumpAnim.onfinish = () => {
        pawn.style.transform = `translate3d(${endX}px, ${endY}px, ${isSuccess ? 0 : -50}px)`;
        
        if (!isSuccess) {
          toast("哎呀，力度没掌握好，掉下去了！");
          pawn.style.opacity = 0; // 掉下深渊
          setTimeout(() => {
            isJumping = false;
            initStage(); // 重置当前关
          }, 800);
          return;
        }

        // 成功跳跃
        const roomData = stepsData[currentStep][selectedTarget.side];
        handleRoomArrival(roomData);
      };
    }

    function handleRoomArrival(roomData) {
      const isFragment = !!roomData.fragment;
      
      if (isFragment) {
        cs.data.collected = cs.data.collected || [];
        cs.data.collected.push(roomData.fragment);
        persist();
      }

      openOverlayHtml(`
        <div class="modal">
          <div class="modal__header">
            <div class="modal__title">抵达：${escapeHtml(roomData.id)}</div>
          </div>
          <div class="modal__body">
            <div class="hintbox" style="font-size:16px; line-height:1.6; text-align:center;">
              ${escapeHtml(roomData.text)}
            </div>
            ${isFragment ? `
              <div style="margin-top:20px; text-align:center; animation: scale-in 0.5s;">
                <div class="badge">获得快乐密钥</div>
                <div style="font-size:36px; color:#a855f7; font-weight:bold; margin-top:8px; text-shadow:0 0 15px rgba(168,85,247,0.5);">[ ${escapeHtml(roomData.fragment)} ]</div>
              </div>
            ` : ''}
            <div class="row" style="margin-top:24px; justify-content:center;">
              <button class="btn btn-primary" type="button" id="btnContinueJump">继续漫游</button>
            </div>
          </div>
        </div>
      `);

      document.getElementById("btnContinueJump").addEventListener("click", () => {
        closeOverlay();
        currentStep++;
        isJumping = false;
        renderCase(caseDef.id); // 重新渲染更新下方槽位
      });
    }

    // 绑定事件 (支持鼠标和触摸屏)
    btnCharge.addEventListener("mousedown", startCharge);
    btnCharge.addEventListener("touchstart", startCharge, { passive: false });
    window.addEventListener("mouseup", endCharge);
    window.addEventListener("touchend", endCharge);

    // 终端与重置逻辑
    document.getElementById("btnRestart5")?.addEventListener("click", () => {
      cs.data.collected = []; // 清空收集
      persist();
      renderCase(caseDef.id);
    });

    document.getElementById("btnSubmit5")?.addEventListener("click", () => {
      const input = (document.getElementById("case5Input")?.value || "").trim().toUpperCase();
      const expected = String(caseDef.solution?.password || "HAPPY").toUpperCase();
      
      if (input === expected) {
        cs.data.fragment = expected;
        persist();
        completeCase(caseDef, caseDef.solution?.revealText);
      } else {
        registerWrongAttempt(cs, "密钥不对。请检查拼写，或者点击【重新漫游】重新收集！");
      }
    });

    // 初始化场景
    initStage();
  }
```

### 💡 这套代码做到了什么？

1. **绝对视觉惊艳**：我没有用让浏览器变形的原生 3D 旋转，而是手写了等距透视（Isometric）容器。里面的踏板是带有霓虹光晕的高级黑科技卡片，而文字我通过 `transform: rotate(-45deg) scaleY(2)` 做了**完美逆向抗畸变**，文字永远是竖直且清晰的。
2. **真正的跳一跳手感**：不再是那种来回跑的进度条。现在是你**按住按钮**，进度条快速增长，小黄人会被**“压扁”**蓄力；你**松开手**，小黄人会沿着计算好的**抛物线 (Cubic Bezier)** 飞在空中！
3. **安全与容错**：如果你按得太久或者太短（没有落在绿色区域），小黄人会掉下去消失，但是**不会扣分，直接原点复活**，让游戏保持挑战性但毫无挫败感。
4. **HAPPY 完美收集**：跳中正确的房间，会弹出发光的盲盒卡片，提取出的 `H`, `A`, `P`, `P`, `Y` 字母会立刻填入下方专门为你写的 5 个精美霓虹收集槽中！
